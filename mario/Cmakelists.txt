cmake_minimum_required(VERSION 3.15)
project(mario VERSION 1.0.0 DESCRIPTION "It's a me Mario" LANGUAGES CXX)

# Set the path variables.
set(CMAKE_EXPORT_COMPILE_COMMANDS true)

# set the include directory
include_directories(include)

# Define the actual executables
add_executable(mario src/mario.cpp)

# Define the testing stuff
enable_testing()
find_package(GTest REQUIRED)
include_directories(${GTEST_INCLUDE_DIRS})

add_executable(test_user_input test/test_user_input.cpp)
target_link_libraries(test_user_input GTest::GTest GTest::Main)

gtest_discover_tests(test_user_input)
